72  std::vector<std::string> Wrap(const std::string &in, const std::string::size_type width, std::string::size_type firstlinewidth = 0)
73  {
74  // Preserve existing line breaks
75  const std::string::size_type newlineloc = in.find('\n');
76  if (newlineloc != in.npos)
77  {
78  std::vector<std::string> first(Wrap(std::string(in, 0, newlineloc), width));
79  std::vector<std::string> second(Wrap(std::string(in, newlineloc + 1), width));
80  first.insert(
81  std::end(first),
82  std::make_move_iterator(std::begin(second)),
83  std::make_move_iterator(std::end(second)));
84  return first;
85  }
86  if (firstlinewidth == 0)
87  {
88  firstlinewidth = width;
89  }
90  std::string::size_type currentwidth = firstlinewidth;
91 
92  std::istringstream stream(in);
93  std::vector<std::string> output;
94  std::ostringstream line;
95  std::string::size_type linesize = 0;
96  while (stream)
97  {
98  std::string item;
99  stream >> item;
100  std::string::size_type itemsize = Glyphs(item);
101  if ((linesize + 1 + itemsize) > currentwidth)
102  {
103  if (linesize > 0)
104  {
105  output.push_back(line.str());
106  line.str(std::string());
107  linesize = 0;
108  currentwidth = width;
109  }
110  }
111  if (itemsize > 0)
112  {
113  if (linesize)
114  {
115  ++linesize;
116  line << " ";
117  }
118  line << item;
119  linesize += itemsize;
120  }
121  }
122  if (linesize > 0)
123  {
124  output.push_back(line.str());
125  }
126  return output;
127  }

921  void Help(std::ostream &help) const
922  {
923  bool hasoptions = false;
924  bool hasarguments = false;
925 
926  const std::vector<std::string> description(Wrap(this->description, helpParams.width - helpParams.descriptionindent));
927  const std::vector<std::string> epilog(Wrap(this->epilog, helpParams.width - helpParams.descriptionindent));
928  std::ostringstream prognameline;
929  prognameline << prog;
930  if (HasFlag())
931  {
932  hasoptions = true;
933  prognameline << " {OPTIONS}";
934  }
935  for (const std::string &posname: GetPosNames())
936  {
937  hasarguments = true;
938  prognameline << " [" << posname << ']';
939  }
940  const std::vector<std::string> proglines(Wrap(prognameline.str(), helpParams.width - (helpParams.progindent + 4), helpParams.width - helpParams.progindent));
941  auto progit = std::begin(proglines);

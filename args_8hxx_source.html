85  std::vector<std::string> Wrap(const std::string &in, const std::string::size_type width, std::string::size_type firstlinewidth = 0)
86  {
87  // Preserve existing line breaks
88  const std::string::size_type newlineloc = in.find('\n');
89  if (newlineloc != in.npos)
90  {
91  std::vector<std::string> first(Wrap(std::string(in, 0, newlineloc), width));
92  std::vector<std::string> second(Wrap(std::string(in, newlineloc + 1), width));
93  first.insert(
94  std::end(first),
95  std::make_move_iterator(std::begin(second)),
96  std::make_move_iterator(std::end(second)));
97  return first;
98  }
99  if (firstlinewidth == 0)
100  {
101  firstlinewidth = width;
102  }
103  std::string::size_type currentwidth = firstlinewidth;
104 
105  std::istringstream stream(in);
106  std::vector<std::string> output;
107  std::ostringstream line;
108  std::string::size_type linesize = 0;
109  while (stream)
110  {
111  std::string item;
112  stream >> item;
113  std::string::size_type itemsize = Glyphs(item);
114  if ((linesize + 1 + itemsize) > currentwidth)
115  {
116  if (linesize > 0)
117  {
118  output.push_back(line.str());
119  line.str(std::string());
120  linesize = 0;
121  currentwidth = width;
122  }
123  }
124  if (itemsize > 0)
125  {
126  if (linesize)
127  {
128  ++linesize;
129  line << " ";
130  }
131  line << item;
132  linesize += itemsize;
133  }
134  }
135  if (linesize > 0)
136  {
137  output.push_back(line.str());
138  }
139  return output;
140  }

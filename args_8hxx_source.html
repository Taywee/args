63  std::vector<std::string> Wrap(const std::string &in, const std::string::size_type width, std::string::size_type firstlinewidth = 0)
64  {
65  // Preserve existing line breaks
66  const std::string::size_type newlineloc = in.find('\n');
67  if (newlineloc != in.npos)
68  {
69  std::vector<std::string> first(Wrap(std::string(in, 0, newlineloc), width));
70  std::vector<std::string> second(Wrap(std::string(in, newlineloc + 1), width));
71  first.insert(
72  std::end(first),
73  std::make_move_iterator(std::begin(second)),
74  std::make_move_iterator(std::end(second)));
75  return first;
76  }
77  if (firstlinewidth == 0)
78  {
79  firstlinewidth = width;
80  }
81  std::string::size_type currentwidth = firstlinewidth;
82 
83  std::istringstream stream(in);
84  std::vector<std::string> output;
85  std::ostringstream line;
86  std::string::size_type linesize = 0;
87  while (stream)
88  {
89  std::string item;
90  stream >> item;
91  std::string::size_type itemsize = Glyphs(item);
92  if ((linesize + 1 + itemsize) > currentwidth)
93  {
94  if (linesize > 0)
95  {
96  output.push_back(line.str());
97  line.str(std::string());
98  linesize = 0;
99  currentwidth = width;
100  }
101  }
102  if (linesize)
103  {
104  ++linesize;
105  line << " ";
106  }
107  line << item;
108  linesize += itemsize;
109  }
110  if (linesize > 0)
111  {
112  output.push_back(line.str());
113  }
114  return output;
115  }

758  std::string Help(unsigned int width = 80, unsigned int progindent = 2, unsigned int descriptionindent = 4, unsigned int flagindent = 6, unsigned int helpindent = 25, unsigned int gutter = 1) const
759  {
760  bool hasoptions = false;
761  bool hasarguments = false;
762 
763  const std::vector<std::string> description(Wrap(this->description, width - descriptionindent));
764  const std::vector<std::string> epilog(Wrap(this->epilog, width - descriptionindent));
765  std::ostringstream help;
766  std::ostringstream prognameline;
767  prognameline << prog;
768  if (HasFlag())
769  {
770  hasoptions = true;
771  prognameline << " {OPTIONS}";
772  }
773  for (const std::string &posname: GetPosNames())
774  {
775  hasarguments = true;
776  prognameline << " [" << posname << ']';
777  }
778  const std::vector<std::string> proglines(Wrap(prognameline.str(), width - (progindent + 4), width - progindent));
779  auto progit = std::begin(proglines);
780  if (progit != std::end(proglines))
781  {
782  help << std::string(progindent, ' ') << *progit << '\n';
783  ++progit;
784  }
785  for (; progit != std::end(proglines); ++progit)
786  {
787  help << std::string(progindent + 4, ' ') << *progit << '\n';
788  }
789 
790  help << '\n';
791 
792  for (const std::string &line: description)
793  {
794  help << std::string(descriptionindent, ' ') << line << "\n";
795  }
796  help << "\n";
797  help << std::string(progindent, ' ') << "OPTIONS:\n\n";
798  for (const auto &description: GetChildDescriptions(shortprefix, longprefix))
799  {
800  const std::string &flags = std::get<0>(description);
801  const std::vector<std::string> info(Wrap(std::get<1>(description), width - helpindent));
802  help << std::string(flagindent, ' ') << flags;
803  auto infoit = std::begin(info);
804  const std::string::size_type flagssize = Glyphs(flags);
805  if ((flagindent + flagssize + gutter) > helpindent)
806  {
807  help << '\n';
808  } else if (infoit != std::end(info))
809  {
810  help << std::string(helpindent - (flagindent + flagssize), ' ') << *infoit << '\n';
811  ++infoit;
812  }
813  for (; infoit != std::end(info); ++infoit)
814  {
815  help << std::string(helpindent, ' ') << *infoit << '\n';
816  }
817  }
818  if (hasoptions && hasarguments)
819  {
820  for (const std::string &item: Wrap(std::string("\"") + terminator + "\" can be used to terminate flag options and force all following arguments to be treated as positional options", width - flagindent))
821  {
822  help << std::string(flagindent, ' ') << item << '\n';
823  }
824  }
825 
826  help << "\n";
827  for (const std::string &line: epilog)
828  {
829  help << std::string(descriptionindent, ' ') << line << "\n";
830  }
831  return help.str();
832  }

838  template <typename T>
839  void ParseArgs(const T &args)
840  {
841  bool terminated = false;
842 
843  // Check all arg chunks
844  for (auto it = std::begin(args); it != std::end(args); ++it)
845  {
846  const std::string &chunk = *it;
847 
848  if (!terminated and chunk == terminator)
849  {
850  terminated = true;
851  // If a long arg was found
852  } else if (!terminated && chunk.find(longprefix) == 0 && chunk.size() > longprefix.size())
853  {
854  const std::string argchunk(chunk.substr(longprefix.size()));
855  // Try to separate it, in case of a separator:
856  const auto separator = argchunk.find(longseparator);
857  const std::string arg = (separator != argchunk.npos ?
858  std::string(argchunk, 0, separator)
859  : argchunk);
860 
861  FlagBase *base = Match(arg);
862  if (base)
863  {
864  ArgFlagBase *argbase = dynamic_cast<ArgFlagBase *>(base);
865  if (argbase)
866  {
867  if (separator != argchunk.npos)
868  {
869  argbase->ParseArg(argchunk.substr(separator + longseparator.size()));
870  } else
871  {
872  ++it;
873  if (it == std::end(args))
874  {
875  std::ostringstream problem;
876  problem << "Argument " << arg << " requires an argument but received none";
877  throw ParseError(problem.str().c_str());
878  } else
879  {
880  argbase->ParseArg(*it);
881  }
882  }
883  } else if (separator != argchunk.npos)
884  {
885  std::ostringstream problem;
886  problem << "Passed an argument into a non-argument flag: " << chunk;
887  throw ParseError(problem.str().c_str());
888  }
889  } else
890  {
891  std::ostringstream problem;
892  problem << "Argument could not be matched: " << arg;
893  throw ParseError(problem.str().c_str());
894  }
895  // Check short args
896  } else if (!terminated && chunk.find(shortprefix) == 0 && chunk.size() > shortprefix.size())
897  {
898  std::string argchunk(chunk.substr(shortprefix.size()));
899  for (auto argit = std::begin(argchunk); argit != std::end(argchunk); ++argit)
900  {
